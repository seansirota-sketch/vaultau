// course.js - Client-side validation BEFORE sending

class QuestionGeneratorUI {
  constructor() {
    this.isGenerating = false;
    this.requestCount = 0;
    this.dailyLimit = 10;
  }

  async generateQuestion(selectedExam) {
    // STEP 1: Validate state
    if (this.isGenerating) {
      showError('Already generating. Please wait.');
      return;
    }

    if (this.requestCount >= this.dailyLimit) {
      showError(`Daily limit (${this.dailyLimit}) reached. Try tomorrow.`);
      return;
    }

    // STEP 2: Validate input
    if (!selectedExam || !selectedExam.id) {
      showError('Please select an exam first.');
      return;
    }

    if (selectedExam.id.length > 100) {
      showError('Invalid exam selection.');
      return;
    }

    // STEP 3: Validate user data
    const user = firebase.auth().currentUser;
    if (!user) {
      showError('Please log in first.');
      return;
    }

    // STEP 4: UI feedback
    this.isGenerating = true;
    showLoadingState();
    this.requestCount++;
    updateQuotaDisplay(this.requestCount, this.dailyLimit);

    try {
      // STEP 5: Call edge function
      const idToken = await user.getIdToken();
      
      const response = await fetch('/api/generate-question', {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${idToken}`,
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({
          examId: selectedExam.id,
          courseId: selectedExam.courseId,
          difficulty: 'medium'  // Could be user-selected
        })
      });

      // STEP 6: Handle response
      if (!response.ok) {
        throw new Error(`HTTP ${response.status}`);
      }

      // STEP 7: Stream response
      const reader = response.body.getReader();
      const decoder = new TextDecoder();
      
      while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        
        const chunk = decoder.decode(value);
        appendToQuestion(chunk);  // Stream to UI
      }

    } catch (error) {
      showError(`Failed to generate: ${error.message}`);
    } finally {
      this.isGenerating = false;
      hideLoadingState();
    }
  }
}

// netlify/edge-functions/generate-question.js - Main orchestrator

export default async (request, context) => {
  const url = new URL(request.url);

  // CORS
  if (request.method === 'OPTIONS') {
    return new Response(null, {
      status: 204,
      headers: {
        'Access-Control-Allow-Origin': '*',
        'Access-Control-Allow-Methods': 'POST',
        'Access-Control-Allow-Headers': 'Content-Type, Authorization'
      }
    });
  }

  try {
    // STEP 1: Parse request
    const body = await request.json();
    const authHeader = request.headers.get('Authorization');

    // STEP 2: Verify auth token
    const uid = await verifyFirebaseToken(authHeader);
    if (!uid) {
      return errorResponse(401, 'Unauthorized');
    }

    // STEP 3: Extract + validate inputs
    const { examId, courseId, difficulty } = body;
    
    if (!examId || !courseId) {
      return errorResponse(400, 'Missing examId or courseId');
    }

    if (examId.length > 100 || courseId.length > 100) {
      return errorResponse(400, 'Invalid input length');
    }

    // STEP 4: Check user quota
    const quota = await checkUserQuota(uid);
    if (quota.remaining <= 0) {
      return errorResponse(429, `Quota exceeded. Reset at ${quota.resetAt}`);
    }

    // STEP 5: Check cache
    const cacheKey = `${courseId}:${examId}:${difficulty}`;
    const cached = await getCachedQuestion(cacheKey);
    
    if (cached && !isExpired(cached.timestamp)) {
      // Return cached result
      return new Response(cached.content, {
        headers: { 'Content-Type': 'text/event-stream' },
        status: 200
      });
    }

    // STEP 6: Choose API (smart fallback)
    const apiChoice = await chooseAPI();
    // Gemini (default): Fast, cheap
    // Claude (fallback): Reliable backup

    // STEP 7: Call API with streaming
    const stream = await generateQuestionStream(
      apiChoice,
      { examId, courseId, difficulty }
    );

    // STEP 8: Cache + stream response
    const encoder = new TextEncoder();
    const transformStream = new TransformStream({
      async transform(chunk, controller) {
        // Stream to client
        controller.enqueue(chunk);
        // Accumulate for cache
      }
    });

    // STEP 9: Log usage
    await logUsage({
      uid,
      examId,
      courseId,
      api: apiChoice,
      timestamp: new Date(),
      cached: false
    });

    return new Response(stream, {
      headers: { 'Content-Type': 'text/event-stream' },
      status: 200
    });

  } catch (error) {
    console.error('Error:', error);
    return errorResponse(500, 'Internal server error');
  }
};

// Helper functions

async function verifyFirebaseToken(authHeader) {
  if (!authHeader?.startsWith('Bearer ')) return null;
  
  const token = authHeader.substring(7);
  
  try {
    const response = await fetch(
      `https://identitytoolkit.googleapis.com/v1/accounts:lookup?key=${Deno.env.get('FIREBASE_API_KEY')}`,
      {
        method: 'POST',
        body: JSON.stringify({ idToken: token })
      }
    );
    
    const data = await response.json();
    return data.users?.[0]?.localId || null;
  } catch {
    return null;
  }
}

async function checkUserQuota(uid) {
  const db = firebase.firestore();
  
  try {
    const snap = await db.collection('user_quotas').doc(uid).get();
    const data = snap.data() || { requests_today: 0, reset_at: tomorrow() };
    
    const remaining = Math.max(0, 10 - data.requests_today);
    
    return {
      used: data.requests_today,
      remaining,
      resetAt: data.reset_at
    };
  } catch (error) {
    console.warn('Quota check failed, defaulting to allow:', error);
    return { remaining: 10 };  // Graceful degradation
  }
}

async function getCachedQuestion(cacheKey) {
  const db = firebase.firestore();
  
  try {
    const snap = await db.collection('ai_questions_cache').doc(cacheKey).get();
    return snap.data() || null;
  } catch {
    return null;
  }
}

async function chooseAPI() {
  // Simple logic: alternate or check Gemini health
  // Could be: Gemini 80% of time, Claude 20% (backup)
  
  const rand = Math.random();
  
  if (rand < 0.8) {
    // Check Gemini health
    const healthy = await isGeminiHealthy();
    return healthy ? 'gemini' : 'claude';
  }
  
  return 'claude';
}

async function generateQuestionStream(api, params) {
  if (api === 'gemini') {
    return generateWithGemini(params);
  } else {
    return generateWithClaude(params);
  }
}

async function generateWithGemini({ examId, courseId, difficulty }) {
  const prompt = buildPrompt(examId, courseId, difficulty);
  
  const response = await fetch(
    `https://generativelanguage.googleapis.com/v1beta/models/gemini-3.1-pro-preview:streamGenerateContent?alt=sse`,
    {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'X-Goog-Api-Key': Deno.env.get('GEMINI_API_KEY')
      },
      body: JSON.stringify({
        contents: [{ parts: [{ text: prompt }] }],
        generationConfig: {
          temperature: 0.65,
          maxOutputTokens: 16384
        }
      })
    }
  );
  
  return response.body;
}

async function generateWithClaude({ examId, courseId, difficulty }) {
  // Fallback using Claude API
  // Same prompt, different endpoint
  
  const prompt = buildPrompt(examId, courseId, difficulty);
  
  const response = await fetch('https://api.anthropic.com/v1/messages', {
    method: 'POST',
    headers: {
      'x-api-key': Deno.env.get('ANTHROPIC_API_KEY'),
      'content-type': 'application/json'
    },
    body: JSON.stringify({
      model: 'claude-opus',
      max_tokens: 4096,
      stream: true,
      messages: [{
        role: 'user',
        content: prompt
      }]
    })
  });
  
  return response.body;
}

function buildPrompt(examId, courseId, difficulty) {
  return `
Generate a high-quality exam question for:
- Exam ID: ${examId}
- Course: ${courseId}
- Difficulty: ${difficulty}

Requirements:
1. Question should be clear and specific
2. Provide 4 multiple-choice options (A, B, C, D)
3. Include the correct answer
4. Write explanation for why it's correct
5. Keep Hebrew context in mind

Format as JSON:
{
  "question": "...",
  "options": ["A: ...", "B: ...", "C: ...", "D: ..."],
  "correct_answer": "A",
  "explanation": "..."
}
  `;
}

async function logUsage(data) {
  const db = firebase.firestore();
  
  try {
    await db.collection('generate_usage').add(data);
    
    // Update user quota
    await db.collection('user_quotas').doc(data.uid).update({
      requests_today: firebase.firestore.FieldValue.increment(1),
      last_request: new Date()
    });
  } catch (error) {
    console.error('Failed to log usage:', error);
    // Don't fail request for logging error
  }
}

function errorResponse(status, message) {
  return new Response(
    JSON.stringify({ error: message }),
    { status, headers: { 'Content-Type': 'application/json' } }
  );
}

Quota System (Per User)

FIRESTORE SCHEMA:

collection('user_quotas')
├─ doc(uid)
│  ├─ requests_today: 10  (max daily)
│  ├─ reset_at: 2025-03-24T00:00:00Z
│  ├─ requests_this_week: 50  (max weekly)
│  ├─ is_premium: false
│  ├─ premium_limit: null
│  └─ last_request: 2025-03-23T14:32:00Z

QUOTA TIERS:
├─ Free: 10 requests/day, 50/week
├─ Premium: 100 requests/day, 500/week
└─ Enterprise: Unlimited

RESET LOGIC:
├─ Daily: Midnight in user's timezone
├─ Weekly: Sunday midnight
├─ Monthly: 1st of month

Caching Strategy

CACHE KEY STRUCTURE:

{courseId}:{examId}:{difficulty}:{version}

Example:
math-101:final-2024:hard:v1

CACHE CONTENT:
├─ question: "What is the derivative of..."
├─ options: ["A: ...", "B: ...", "C: ...", "D: ..."]
├─ correct_answer: "A"
├─ explanation: "Because..."
├─ generated_at: 2025-03-23T14:00:00Z
├─ expires_at: 2025-03-23T20:00:00Z (6 hours TTL)
├─ hit_count: 42
└─ generated_by: "gemini"

PRE-GENERATION (Background Job):
├─ Run daily at 2 AM
├─ Target popular courses (top 10)
├─ Generate 5 questions per exam
├─ Cost: $0.50-1.00 per day
└─ Benefit: Instant responses during peak hours

Error Handling & Fallback

ERROR SCENARIOS:

1. Gemini API Error
   ├─ 429 (Rate Limited)
   │  └─ Wait + retry with backoff
   ├─ 500 (Server Error)
   │  └─ Fall back to Claude
   └─ 403 (Quota Exceeded)
      └─ Return cached or error

2. Claude API Error
   └─ Both APIs down?
      └─ Return "Please try again later"

3. Firebase Auth Error
   └─ Can't verify token
      └─ Reject with 401

4. Firestore Error
   ├─ Cache write fails
   │  └─ Log error, don't break response
   └─ Quota check fails
      └─ Allow request (degrade gracefully)

RETRY STRATEGY:
├─ Immediate retry: 2 times
├─ Exponential backoff: 100ms, 200ms
├─ Max retry time: 5 seconds
└─ Then fall back or error

Cost Tracking

COST MODEL:

Gemini API:
├─ Input: $0.001 per 1M tokens
├─ Output: $0.002 per 1M tokens
├─ Typical question: 500 input + 1000 output tokens
├─ Cost per question: ~$0.0025
└─ 1000 questions/day: ~$2.50

Claude API (Fallback):
├─ Input: $0.003 per 1M tokens
├─ Output: $0.015 per 1M tokens
├─ Same 500+1000 tokens
├─ Cost per question: ~$0.0185
└─ Use sparingly (backup only)

CACHING SAVINGS:
├─ Hit rate target: 70%
├─ Cost reduction: 70%
├─ Actual cost per question: ~$0.00075
└─ Monthly saving: 70% of API costs

MONITORING:
